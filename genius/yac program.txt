save this file with following file name expr.y



%{
#include <stdio.h>
#include <stdlib.h>

int yylex();
void yyerror(const char *s);
%}

%token NUMBER

%left '+' '-'
%left '*' '/'

%%
/* Grammar rules for expression parsing and evaluation */
E : E '+' E    { $$ = $1 + $3; }
  | E '-' E    { $$ = $1 - $3; }
  | E '*' E    { $$ = $1 * $3; }
  | E '/' E    {
                    if ($3 == 0)
                        yyerror("Division by zero");
                    else
                        $$ = $1 / $3;
                }
  | '(' E ')'  { $$ = $2; }
  | NUMBER     { $$ = $1; }
  ;
%%

int main() {
    printf("Enter an expression: ");
    if (yyparse() == 0)
        printf("Expression is valid.\n");
    return 0;
}

void yyerror(const char *s) {
    printf("Invalid expression: %s\n", s);
}
